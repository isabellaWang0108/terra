import { Meta } from '@storybook/addon-docs'
import { Card } from '@sb-config/components/Card'
import { Tabs, Tab } from '@sb-config/components/Tabs'
import TypographyOverview from './TypographyOverview.mdx'
import TypographyUsage from './TypographyUsage.mdx'

export const TypographyPage = () => {
  const [currentTab, setCurrentTab] = React.useState('overview')
  return (
    <>
      <Tabs>
        <Tab
          id={'overview'}
          onClick={() => {
            setCurrentTab('overview')
          }}
          selected={currentTab === 'overview'}
        >
          Overview
        </Tab>
        <Tab
          id={'usage'}
          onClick={() => {
            setCurrentTab('usage')
          }}
          selected={currentTab === 'usage'}
        >
          Usage
        </Tab>
        <Tab
          id={'code'}
          onClick={() => {
            setCurrentTab('code')
          }}
          selected={currentTab === 'code'}
        >
          Code
        </Tab>
      </Tabs>
      {currentTab === 'overview' ? <TypographyOverview /> : null}
      {currentTab === 'usage' ? <TypographyUsage /> : null}
    </>
  )
}
