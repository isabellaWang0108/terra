export const TypographyPage = () => {

  return (

          id={'overview'}
          onClick={() => {

          }}
          selected={currentTab === 'overview'}
        >
          Overview
        </Tab>
        <Tab
          id={'usage'}
          onClick={() => {
            setCurrentTab('usage')
          }}
          selected={currentTab === 'usage'}
        >
          Usage
        </Tab>
        <Tab
          id={'code'}
          onClick={() => {
            setCurrentTab('code')
          }}
          selected={currentTab === 'code'}
        >
          Code
        </Tab>
      </Tabs>
      {currentTab === 'overview' ? <TypographyOverview /> : null}
      {currentTab === 'usage' ? <TypographyUsage /> : null}
    </>
  )
}
